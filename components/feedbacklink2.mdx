export const FeedbackLink = () => {
  const [currentUrl, setCurrentUrl] = React.useState('');

  React.useEffect(() => {
    // Captures the URL once the component mounts in the browser
    setCurrentUrl(encodeURIComponent(window.location.href));
  }, []);

  const formBase = "https://docs.google.com/forms/d/e/YOUR_FORM_ID/viewform?usp=pp_url";
  const entryId = "entry.123456789"; // Replace with your ID
  
  return (
    <a 
      href={`${formBase}&${entryId}=${currentUrl}`}
      target="_blank"
      className="feedback-button"
    >
      Report an issue on this page
    </a>
  );
};
